- Trên máy ảo centos 6 hoặc 7 đã cài đặt, tiến hành cài đặt mô hình:
  - LAMP stack :  build wordpress webiste trên chính mô hình trên
  - LEMP stack : build wordpress webiste trên chính mô hình trên

trên mỗi mô hình 2 đứa build giúp anh 2 web (tham khảo từ khoá Vhost, vhost là gì, ứng dụng của vhost trong web server)
wordpress.training.vn
laravel.training.vn
cài wordpress và laravel vào mỗi domain nhá

thao tác trên máy ảo, thử trỏ /etc/hosts xem domain có nhận được đúng không


server {
    listen 80;
    server_name laravel.training.vn www.laravel.training.vn;
    root /var/www/html/laravel.training.vn/public;
 
    index index.html index.htm index.php;
 
    charset utf-8;
 
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }
 
    location = /favicon.ico { access_log off; log_not_found off; }
    location = /robots.txt  { access_log off; log_not_found off; }
 
    error_page 404 /index.php;
 
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php8.1-fpm.sock;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
    }
 
    location ~ /\.(?!well-known).* {
        deny all;
    }
}


server {
